## Installation

```bash
npx changeelog/reactuse@latest add useRandom
```

## Usage

```ts
const { value, generate, reset, setRange, toggleInteger } = useRandom({
  min: 1,
  max: 10,
  isInteger: true,
});

console.log(value); // null (initial value)
console.log(generate()); // Random integer between 1 and 10
setRange(0, 100);
toggleInteger();
console.log(generate()); // Random float between 0 and 100
reset();
```
