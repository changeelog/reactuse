---
title: useClamp
description: Clamping a number between a minimum and maximum value
---

## Installation

```bash
npx changeelog/reactuse@latest add useClamp
```

## Usage

```tsx
import { useClamp } from '@changeelog/reactuse';

function ClampedValueDisplay() {
  const value = 15;
  const min = 0;
  const max = 10;
  const clampedValue = useClamp(value, min, max);

  return (
    <p>
      The value {value} clamped between {min} and {max} is {clampedValue}
    </p>
  );
}
```
