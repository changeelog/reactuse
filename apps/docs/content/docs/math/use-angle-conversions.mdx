Demo

## Installation

```bash
npx changeelog/reactuse@latest add useAngleConversions
```

## Usage

```tsx
import { useAngleConversions } from '@changeelog/reactuse';

function AngleConverter() {
  const { radiansToDegrees, degreesToRadians } = useAngleConversions();

  const radians = 1 as Radians;
  const degrees = 180 as Degrees;

  return (
    <div>
      <p>
        {radians} radians is equal to {radiansToDegrees(radians)} degrees
      </p>
      <p>
        {degrees} degrees is equal to {degreesToRadians(degrees)} radians
      </p>
    </div>
  );
}
```
